üöÄ Quick Commands

Get Visualization URLs

# Get localhost URL
scripts/run.sh scripts/get_visualization_url.py example

# Get network URL (for Fire TV)
scripts/run.sh scripts/get_visualization_url.py example --network

Display on Fire TV

# Open visualization on Fire TV via ADB
adb shell am start -a android.intent.action.VIEW \
-d "http://192.168.0.28:5173/visualizations/example.html"

Create New Visualization

# Create new HTML file
touch frontend/public/visualizations/my-viz.html

# Access immediately (no build needed!)
http://localhost:5173/visualizations/my-viz.html
